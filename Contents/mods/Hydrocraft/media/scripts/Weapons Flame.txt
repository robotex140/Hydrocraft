recipe Make Soap Shards
    {
       	Soap2/HCSoap,
		keep [Recipe.GetItemTypes.SharpKnife],
       	Result:HCSoapshards,
       	Time:150.0,
       	Category:Chemistry,
    	OnGiveXP:HCCooking_OnGiveXP,
    }

recipe Make Pot With Soap
    {
       	HCSoapshards,
