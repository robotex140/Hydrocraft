/************************RECIPES************************/	

recipe Make Desktop Computer
    {
	SkillRequired:Electricity=1,
        HCComputermonitor,
        HCComputer,
	HCComputerkeyboard,
	HCComputermouse,
	HCMousepad,
	HCPowercord,
	HCTablewood,
	Generator,
        CanBeDoneFromFloor:true,
        NeedToBeLearn:true,
	Result:HCComputerdesktop,
	Time:1000,
        Category:Electrical,
    	OnGiveXP:HCElectricity_OnGiveXP,
    }
