recipe Make Gem Cutter Wheel
    {
	SkillRequired:Woodwork=8;Mechanics=2;,
	HCMagnifyglass,
	HCSilkcloth,
	Tongs,
	Rope,
	HCGearsteel,
	Plank=5,
	Nails=8,
	HCWoodbeam,
	HCStonewheel,
    	keep Hammer,
	keep Saw,
	OnTest:HCNearCarpybench,
    	CanBeDoneFromFloor:true,
   	Result:HCGemcutter,
    	Sound:PZ_Hammer,
    	Time:500,
    	Category:Jewelry,
    	OnGiveXP:HCMechanicsWoodwork_OnGiveXP,
    }
