/************************RECIPES************************/	

recipe Make Taxidermy Workbench
    {
	HCWorkbench,
        HCBonesaw,
        Scalpel/HCScalpelimprovised,
        HCSewingpinbox,
        /*HCPincushion, removed, as this item no longer spawns in vanilla - may need a better solution*/
        HCPliers,
        HCCardboardbox=2,
        HCBookanatomy,
        HCBookmammalguide,
	Nails,
	keep Hammer,
	keep Saw,
        Result:HCTaxidermyworkbench,
        CanBeDoneFromFloor:true,
        NeedToBeLearn:true,
        Time:400,
    	Sound:PZ_Hammer,
	Category:Engineer,
    	OnGiveXP:HCHunger_OnGiveXP,
    }
