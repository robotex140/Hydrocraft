/************************RECIPES************************/

recipe Cut Off Bottle
    {
    	HCWaterbottleclean,
        keep KitchenKnife/HuntingKnife/HCJackknife/FlintKnife/HCBoxcutter/Scissors/HCElectricmultitoolsawpowered,
    	Result:HCWaterbottlecutoff,
    	Time:50,
    	Category:Engineer,
        OnCreate:recipe_hcgetplastic,
   	OnGiveXP:HCHunger_OnGiveXP,
    }

recipe Recycle Plastic Bottle
    {
        HCTrashbottle,
        keep KitchenKnife/HuntingKnife/HCJackknife/FlintKnife/HCBoxcutter/Scissors/HCElectricmultitoolsawpowered,
        Result:HCPlastic=2,
        Time:50,
    	Category:Engineer,
   	OnGiveXP:HCHunger_OnGiveXP,
    } 

recipe Clean Bottle
    {
    	destroy PopBottleEmpty,
        keep Sponge/HCLoofah/HCSponge/RippedSheets,
        Water,
	Soap2/Bleach/HCHandsoap/HCShampoo/HCSoap/CleaningLiquid2,
    	Result:HCWaterbottleclean,
    	Time:50,
    	Category:Chemistry,
   	OnGiveXP:HCHunger_OnGiveXP,
    }
