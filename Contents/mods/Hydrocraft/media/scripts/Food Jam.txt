recipe Prepare Jam Pot
	{
        	SkillRequired:Cooking=4,
		destroy WaterPot,
		Sugar,
		HCLemonjuicebottle/HCLemonjuicebowl,
		Result:HCJamstarter,
		Time:90.0,
	   	Category:Cooking,
		OnCreate:HCRecipeBowls,
       		OnGiveXP:HCCooking_OnGiveXP,
	}

recipe Prepare Grape Jam
	{
        	SkillRequired:Cooking=4,
		destroy HCJamstarter,
		Grapes=4,
		Result:HCJamgrapepot,
		Time:90.0,
	   	Category:Cooking,
       		OnGiveXP:HCCooking_OnGiveXP,
	}
